// FUNCIONES

// FUNCION QUE NO RECIBE NI RETORNA VALOR
function saludo(){
	echo "Hola ";
}

// FUNCION QUE RECIBE Y NO RETORNA VALOR
function saludo($nombre){
	echo "Hola ".$nombre;
}

// FUNCION QUE NO RECIBE PERO SI RETORNA VALOR
function saludo(){
	return "Hola";
}

// FUNCION QUE RECIBE Y RETORNA VALOR
function saludo($nombre, $apellido){
	$saludo = "Hola ".$nombre." ".$apellido;
	return $saludo;
}
